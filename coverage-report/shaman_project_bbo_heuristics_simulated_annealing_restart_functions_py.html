23def random_restart(bernouilli_parameter, initial_temperature, **kwargs):

26 Returns the system maximal temperature as this type of restart restarts the system completely. 

27 

28 Args: 

29 bernouilli_parameter (float): A float located between 0 and 1. 

30 initial_temperature (float): The system's maximal temperature. 

31 

32 Returns: 

33 tuple of bool and float: Whether or not the system should restart and the system's maximal 

34 temperature. 

35 """ 

36 assert 0 <= bernouilli_parameter <= 1, ( 

37 "Bernouilli's law parameter must be located between 0 " "and 1." 

38 ) 

39 flag = np.random.binomial(1, bernouilli_parameter) 

40 return flag == 1, initial_temperature
