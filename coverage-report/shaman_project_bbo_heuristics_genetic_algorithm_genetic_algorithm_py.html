73 def choose_next_parameter(self, history, ranges, *args, **kwargs): 

74 """ 

75 Select the next parameters for the optimization process, in this case the children of the 

76 two parents selected as the fittest. 

77 

78 A genetic algorithm has the following rule for choosing the next parameter: 

79 1) Use a selection method to pick two parents fit for mating 

80 2) Use a crossover method to mate those two parents 

81 3) Probabilistically determine whether or not the mutation method should be applied. 

82 

83 Args: 

84 history (dict): the history of the optimization, i.e. the tested parameters and the 

85 associated value. 

86 ranges (numpy array of numpy arrays): the possible values of each parameter dimension. 

87 

88 Returns: 

89 numpy array: The next parameter, i.e. the child born from the 

90 reproduction of the two parents. 

91 """ 

92 idx = 0 

93 # loop until the child is different from its two parents 

94 while True and idx < self.max_repeat: 

95 # Select two parents using the selection method 

96 parent_1, parent_2 = self.selection_method( 

97 history=history, *self.args, **self.kwargs 

98 ) 

99 # Mate those two parents to compute a new child 

100 child = self.crossover_method(parent_1, parent_2) 

101 

102 # Is there a mutation at this round? Compute the probability using a bernouilli random 

103 # variable 

104 mutation = np.random.binomial(1, self.mutation_rate) 

105 # If so, perform mutation on the child and return the mutant form 

106 if mutation: 

107 child = self.mutation_method(child, ranges) 

108 self.nbr_mutation += 1 

109 if not np.array_equal(child, parent_1) and not np.array_equal( 

110 child, parent_2 

111 ): 

112 break 

113 idx += 1 

114 self.family_line.append((parent_1, parent_2, child)) 

115 return child
